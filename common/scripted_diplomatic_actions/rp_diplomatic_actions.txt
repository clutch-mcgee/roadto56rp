scripted_diplomatic_actions = {
	
	demilitarize_subject_action = {

		allowed = {
			ROOT = { is_ai = no }
		}

		visible = {
			PREV = {
				AND = {
					NOT = { has_idea = subject_military_restrictions }
					exists = yes
					is_subject_of = ROOT
					is_ai=yes
				}
			}
		}

		selectable = {
			custom_trigger_tooltip = {
				tooltip = RESTRICTING_ALLOWED_TOOLTIP
				always = yes
			}	
		}

		cost = 150

		command_power = 15

		requires_acceptance = no

		icon = 1

		complete_effect = {
			PREV = {
				add_timed_idea = {
					idea = recently_restricted
					days = 20
				}
				add_ideas = subject_military_restrictions
				add_ideas = cant_train_divisions
				country_lock_all_division_template = yes
				delete_unit = {
					disband = yes
				}
			}
		}
	}

	remilitarize_subject_action = {

		allowed = {
			ROOT = { is_ai = no }
		}

		visible = {
			PREV = {
				AND = {
					has_idea = subject_military_restrictions
					exists = yes
					is_subject_of = ROOT
				}
			}
		}

		selectable = {
			IF = {
				limit = {
					PREV = {
						has_idea = recently_restricted
					}
				}
				custom_trigger_tooltip = {
					tooltip = DERESTRICTING_DENIED_TOOLTIP
					always = no
				}
			}
			else = {
				custom_trigger_tooltip = {
					tooltip = DERESTRICTING_ALLOWED_TOOLTIP
					always = yes
				}
			}
		}

		cost = 25

		command_power = 5

		requires_acceptance = no

		icon = 1

		complete_effect = {
			PREV = {
				remove_ideas = subject_military_restrictions
				remove_ideas = cant_train_divisions
				country_lock_all_division_template = no
			}
		}
	}
    eda_puppet_independence = {
        icon = 1
        cost = 75
        command_power = 15
        requires_acceptance = no
        show_acceptance_on_action_button = no
        send_description =  "Y[This.GetName]! have always been a faithful subject, their loyalty has always been towards us. It's time we reward them for it."
		receive_description = puppet_independence_recieve_desc

        allowed = {
			ROOT = { is_ai = no }
		}

		visible = {
			PREV = {
				AND = {
					exists = yes
					is_subject_of = ROOT
				}
			}
		}

		selectable = {
			custom_trigger_tooltip = {
				tootip = EDA_PUPPET_INDEPENDENCE_TOOLTIP
				always = yes
			}
		}

     can_be_sent = { always = yes }
        complete_effect = {
			ROOT = {
				end_puppet = PREV
			}
            if = {
				limit = {
					has_idea = subject_military_restrictions
				}
				remove_ideas = subject_military_restrictions
			}
            if = {
				limit = {
					has_idea = cant_train_divisions
				}
				remove_ideas = cant_train_divisions
			}
		}
	} 
}
